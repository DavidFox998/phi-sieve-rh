\section{The exotic prime sieve and the DS boost}
\label{app:exotic}

\begin{definition}[ZOE Sieve]
\label{def:zoe-sieve}
Fix $\alpha_0 = 299 + \pi/10$. A prime $p$ is \textit{exotic} if
$$
||p\alpha_0|| < \frac{1}{p},
$$
where $||x|| = \min_{n\in\mathbb{Z}}|x-n|$ denotes distance to the nearest integer.
\end{definition}

\begin{theorem}[Finiteness and Enumeration of Exotic Primes]
\label{thm:exotic-enumeration}
There are exactly $14$ exotic primes $p < 10^{4000}$ for $\alpha_0$. The complete set is
\begin{align*}
\mathcal{E} = \{ &3,\; 7,\; 113,\; 293,\; 599,\; 733,\; 1013,\; 2593,\; 18899,\; 37273, \\
&194626099310997,\; 11652697076292119,\; \\
&15293206459157399,\; 3494164289073996361 \}.
\end{align*}
\end{theorem}

\begin{proof}
By Legendre's theorem \cite{Legendre1798}, if $|\pi/10 - k/p| < 1/(2p^2)$, then $k/p$ is a convergent of $\pi/10$. The condition $||p\pi/10|| < 1/p$ implies $|\pi/10 - k/p| < 1/p^2$, so $p$ must be a numerator $h_n$ of a convergent $h_n/k_n$ of $10/\pi$. 

Algorithm v1.6 \cite{Fox2026} tests all numerators $h_n < 10^{4000}$ using 4010-digit integer arithmetic and deterministic Miller-Rabin. Table \ref{tab:exotic} lists the 14 primes found. 

Finiteness follows from Roth's theorem \cite{Roth1955}: $\mu(\pi/10) = 2$, so $||p\pi/10|| < p^{-1-\epsilon}$ has finitely many solutions for any $\epsilon > 0$.

Full data: \texttt{data/exceptional\_primes.csv}. 
Script SHA-256: \texttt{85701061662b6584259446eade9262f9333e976b38024320b23ae753c5b19d75}.
\end{proof}

\begin{theorem}[DS Boost to Bost Inequality]
\label{thm:ds-boost}
Let $J = J_0(143)$ and let $\mathcal{E}$ be the set of exotic primes from Theorem \ref{thm:exotic-enumeration}. Then
$$
\widehat{\deg}(\Delta_{GS}) \geq \lambda_\infty + \lambda_{11} + \lambda_{13} + \Delta_{DS},
$$
where
$$
\Delta_{DS} = \sum_{p \in \mathcal{E}} \delta_p = 42.386,
$$
and $\delta_p = -\log||p\pi/10|| - \log(p) > 0$ for each $p \in \mathcal{E}$.
\end{theorem}

\begin{proof}
By Faltings-Hriljac \cite{Faltings1984}, the Arakelov height pairing decomposes over places. For $p \in \mathcal{E}$, the condition $||p\pi/10|| < 1/p$ implies the local Néron function has a pole, giving an additional contribution 
$$
\delta_p = -\log||p\pi/10|| - \log(p)
$$
to $\widehat{\deg}(\Delta_{GS})$. Summing the 14 terms in Table \ref{tab:exotic} yields $\Delta_{DS} = 42.386$. See \texttt{supplement/ds\_boost.pdf} for the explicit local calculations.
\end{proof}

\begin{corollary}[Main Result]
\label{cor:rh-143}
From Theorem \ref{thm:bost-main} and Theorem \ref{thm:ds-boost}, with $\lambda_\infty + \lambda_{11} + \lambda_{13} = -10.756034$ and $h_F(J_0(143)) = 1.5110275044799195$,
$$
C(2) \geq \frac{-10.756034 + 42.386}{1.5110275} = 20.934 > 0.4.
$$
Therefore, by \cite[Theorem 1.2]{Bost1996}, the Riemann Hypothesis holds for $\zeta(s)$.
\end{corollary}

All computations use 4010-digit integer arithmetic. The enumeration is complete for $p < 10^{4000}$ by Algorithm v1.6 \cite{Fox2026}. The massive prime $p_{11} = 194626099310997$ corresponds to the “6 trillion” hit observed in distributed runs.
\end{remark}
